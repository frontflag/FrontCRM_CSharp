方法 2: 执行验证 SQL
----------------------------------------
在查询工具中执行以下 SQL：

-- 检查所有表
SELECT table_name, 
       CASE WHEN table_name IN ('customerinfo','customercontactinfo','customeraddress',
                                'vendorinfo','vendorcontactinfo',
                                'sellorder','purchaseorder','material')
            THEN '✅ EBS业务表'
            ELSE '其他表'
       END as 类型
FROM information_schema.tables 
WHERE table_schema = 'public'
ORDER BY 类型 DESC, table_name;

-- 检查表数量
SELECT COUNT(*) as 总表数,
       SUM(CASE WHEN table_name IN ('customerinfo','customercontactinfo','customeraddress',
                                    'vendorinfo','vendorcontactinfo',
                                    'sellorder','purchaseorder','material') 
                THEN 1 ELSE 0 END) as EBS表数
FROM information_schema.tables 
WHERE table_schema = 'public';

方法 3: 检查表结构
----------------------------------------
执行以下 SQL 验证表结构：

-- 查看客户表结构
SELECT column_name, data_type, is_nullable
FROM information_schema.columns 
WHERE table_schema = 'public' AND table_name = 'customerinfo'
ORDER BY ordinal_position;

-- 查看索引
SELECT tablename, indexname
FROM pg_indexes 
WHERE schemaname = 'public' 
    AND tablename IN ('customerinfo','vendorinfo','sellorder','purchaseorder')
ORDER BY tablename, indexname;
